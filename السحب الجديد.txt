// ========== دالة التحقق من صحة الرقم حسب الشركة ==========
function validatePhoneNumber(phone, method) {
  let cleanPhone = phone.replace(/\s+/g, '').replace(/[^\d]/g, '');
  const prefixes = {
    'فودافون كاش': ['010', '015', '016'],
    'اتصالات كاش': ['011'],
    'اورنج كاش': ['012']
  };
  if (!prefixes[method]) return true;
  if (cleanPhone.length !== 11) return false;
  return prefixes[method].some(prefix => cleanPhone.startsWith(prefix));
}

async function checkWithdrawConditions(event) {
  const method = document.getElementById("method").value;
  const address = document.getElementById("accountAddress").value.trim();
  const amount = parseFloat(document.getElementById("amount").value);
  const currency = document.querySelector('input[name="currency"]:checked').value;

  // التحقق من البيانات الأساسية
  if (!method || !address || !amount) {
    showCustomToast("يرجى إدخال جميع البيانات", "error");
    return;
  }

  // التحقق من الحد الأدنى
  if (currency === 'USD' && amount < minAmountUSD) {
    showCustomToast(`الحد الأدنى للسحب هو ${minAmountUSD} دولار`, "error");
    return;
  }
  if (currency === 'EGP' && amount < minAmountEGP) {
    showCustomToast(`الحد الأدنى للسحب هو ${minAmountEGP} جنيه مصري`, "error");
    return;
  }

  // التحقق من صحة الرقم
  if (method === "فودافون كاش" || method === "اتصالات كاش" || method === "اورنج كاش") {
    if (!validatePhoneNumber(address, method)) {
      let errorMsg = "";
      if (method === "فودافون كاش") {
        errorMsg = "رقم فودافون كاش يجب أن يبدأ بـ 010 أو 015 أو 016 ويتكون من 11 رقم";
      } else if (method === "اتصالات كاش") {
        errorMsg = "رقم اتصالات كاش يجب أن يبدأ بـ 011 ويتكون من 11 رقم";
      } else if (method === "اورنج كاش") {
        errorMsg = "رقم اورنج كاش يجب أن يبدأ بـ 012 ويتكون من 11 رقم";
      }
      showCustomToast(errorMsg, "error");
      return;
    }
  }

  if (!currentUser) {
    window.location.href = "login.html";
    return;
  }

  // التحقق من الرصيد (مطلوب)
  let amountUSD = amount;
  if (currency === 'EGP') {
    amountUSD = amount / USD_TO_EGP;
  }

  if (amountUSD > userFiatBalance) {
    showCustomToast("رصيد غير كافٍ لإتمام السحب", "error");
    return;
  }

  // تخزين بيانات السحب مؤقتاً
  pendingWithdrawData = {
    method,
    address,
    amount,
    currency,
    amountUSD
  };

  // ✅ إظهار مودال الشرط قبل إتمام السحب
  showConditionModal();
}

// ========== إتمام عملية السحب (تُستدعى بعد استيفاء الشرط) ==========
async function completeWithdraw() {
  if (!pendingWithdrawData || !currentUser) return;

  const { method, address, amount, currency, amountUSD } = pendingWithdrawData;
  const submitBtn = document.getElementById("submitBtn");
  const loader = document.getElementById("loader");

  submitBtn.disabled = true;
  submitBtn.innerText = "جاري المعالجة...";
  loader.style.display = "block";

  try {
    const userDoc = await db.collection("users").doc(currentUser.uid).get();
    if (!userDoc.exists) {
      throw new Error("المستخدم غير موجود");
    }
    const userData = userDoc.data();

    const withdrawData = {
      userId: currentUser.uid,
      username: userData.username || "",
      method: method,
      address: address,
      amount: amount,
      currency: currency,
      amountUSD: amountUSD,
      status: "pending",
      createdAt: firebase.firestore.FieldValue.serverTimestamp()
    };

    await db.collection("withdrawals").add(withdrawData);

    await db.collection("users").doc(currentUser.uid).update({
      fiatBalance: firebase.firestore.FieldValue.increment(-amountUSD)
    });

    // إرسال إشعار إلى تيليجرام
    try {
      const message = `
🆕 طلب سحب جديد
👤 المستخدم: ${userData.username || 'غير معروف'}
🔹 طريقة السحب: ${method}
📞 العنوان/المحفظة: ${address}
💰 المبلغ: ${amount} ${currency}
🆔 معرف المستخدم: ${currentUser.uid}
🕐 الوقت: ${new Date().toLocaleString('ar-EG')}
      `;
      await fetch(`https://api.telegram.org/bot${TELEGRAM_TOKEN}/sendMessage`, {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ chat_id: TELEGRAM_CHAT_ID, text: message, parse_mode: 'HTML' })
      });
    } catch (e) { console.error(e); }

    showCustomToast("تم إرسال طلب السحب بنجاح، سيتم مراجعته قريباً", "success");
    
    document.getElementById("method").value = "";
    document.getElementById("accountAddress").value = "";
    document.getElementById("amount").value = "";
    document.getElementById("convertedAmount").innerHTML = "";

    setTimeout(() => {
      window.location.href = "transactions.html";
    }, 2000);

  } catch (err) {
    console.error(err);
    showCustomToast("حدث خطأ: " + err.message, "error");
    submitBtn.disabled = false;
    submitBtn.innerText = "تأكيد طلب السحب";
    loader.style.display = "none";
  } finally {
    pendingWithdrawData = null;
  }
}

// ========== آلية النقر للـ Navbar ==========
const navItems = document.querySelectorAll('.nav-item');
const currentPath = window.location.pathname.split('/').pop() || 'withdraw.html';
navItems.forEach(item => item.classList.remove('current-page'));
navItems.forEach(item => {
  if (item.dataset.url === currentPath) {
    item.classList.add('current-page');
  }
});

let effectShown = new Array(navItems.length).fill(false);
let timerId = null;
let currentNavItem = null;

function showEffect(item) {
  if (item.classList.contains('current-page')) return;
  
  if (timerId) {
    clearTimeout(timerId);
    timerId = null;
    if (currentNavItem) {
      const oldCircle = currentNavItem.querySelector('.nav-circle');
      const oldEgypt = currentNavItem.dataset.egypt;
      oldCircle.textContent = oldEgypt;
      currentNavItem.classList.remove('showing');
    }
  }
  
  currentNavItem = item;
  const circle = item.querySelector('.nav-circle');
  const altSymbol = item.dataset.alt;
  circle.textContent = altSymbol;
  item.classList.add('showing');
  
  timerId = setTimeout(() => {
    const circle = currentNavItem.querySelector('.nav-circle');
    const egyptSymbol = currentNavItem.dataset.egypt;
    circle.textContent = egyptSymbol;
    currentNavItem.classList.remove('showing');
    timerId = null;
    currentNavItem = null;
  }, 3000);
}

navItems.forEach((item, index) => {
  item.addEventListener('click', function(e) {
    e.stopPropagation();
    
    if (this.classList.contains('current-page')) return;
    
    if (effectShown[index]) {
      window.location.href = this.dataset.url;
      return;
    }
    
    showEffect(this);
    effectShown[index] = true;
  });
});
